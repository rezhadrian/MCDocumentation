100 template<typename _Ty>
101 Point<_Ty> interpolate(const Vertex<_Ty>& vertex1, const Vertex<_Ty>& vertex2, const _Ty isoValue)
102#if MC_VERSION_GREATER_THAN(0, 7)
103 noexcept
104#endif // > v0.7
105 {
106 // Relative distance of interpolating point from the first vertex
107 auto diff = vertex2.value - vertex1.value;
108#if !MC_VERSION_GREATER_THAN(0, 7)
109 api::error::rtcheck(std::abs(diff) > core::epsilon<_Ty>(), api::error::LE, MC_ERROR_MSG("division by zero"));
110#endif // <= v0.7
111 _Ty ratio = (isoValue - vertex1.value) / diff;
112 return (_Ty(1) - ratio) * vertex1.point + ratio * vertex2.point;
113 }

123 template<typename _Ty
124#ifdef MC_GRID_ITERATOR
125 , typename _Data
126#endif // !MC_GRID_ITERATOR
127 >
128 IntersectionEdges<_Ty> getIntersectionCoordinates(const Cube<_Ty
129#ifdef MC_GRID_ITERATOR
130 , _Data
131#endif // !MC_GRID_ITERATOR
132 >& cube, const _Ty isoValue)
133#if MC_VERSION_GREATER_THAN(0, 7)
134 noexcept
135#endif // > v0.7
136 {
137 auto intersectionsKey = tables::IndexToEdges[calculateCubeIndex(cube, isoValue)];
138
139 IntersectionEdges<_Ty> intersections;
140 std::transform(
141 intersections.begin(), intersections.end(), tables::Edges, intersections.begin(),
142 [&](Point<_Ty>& point, const MCindex edgeIndex) {
143 if (intersectionsKey & edgeIndex) {
144 auto idx = core::log2(edgeIndex);
145 return interpolate(
146 cube[tables::EdgeToVertices[idx].first],
147 cube[tables::EdgeToVertices[idx].second],
148 isoValue
149 );
150 }
151 return point;
152 }
153 );
154 return intersections;
155 }

159 * \param cube The target \c Cube;
160 * \param isoValue A value to determine whether a vertex is above the \a isosurface or not;
161 * \tparam _Ty A floating-point type;
162 * \return \ref __mc_polygons__ ["Polygons"] of spatial points of the triangulated \c Cube.

170#ifdef MC_GRID_ITERATOR
171 , _Data
172#endif // !MC_GRID_ITERATOR
173 >& cube, const _Ty isoValue)
174#if MC_VERSION_GREATER_THAN(0, 7)
175 noexcept
176#endif // > v0.7
177 {
178 Polygons<_Ty> triangles;
179
180 IntersectionEdges<_Ty> intersections = getIntersectionCoordinates(cube, isoValue);
181 auto cubeIndex = calculateCubeIndex(cube, isoValue);
182
183 for (MCindex i = 0; tables::EdgesToTriangles[cubeIndex][i] != -1; i += 3) {
184 triangles.push_back({
185 intersections[tables::EdgesToTriangles[cubeIndex][i ]],
186 intersections[tables::EdgesToTriangles[cubeIndex][i + 1]],
187 intersections[tables::EdgesToTriangles[cubeIndex][i + 2]]
188 }
189 );
190 }
191
192 return triangles;

266 struct RegularTriangulator final : public Triangulator<RegularMesh<_Ty>> {
267 using Container = typename Triangulator<RegularMesh<_Ty>>::Container;

269 template<class _Data>
270 MC_CONSTEXPR auto operator()(
271 const Grid<_Data>& grid, const typename _Data::Float isoValue
272 ) const {
273 auto domain = grid.domain();
274 auto size = domain.size();
275 GridSizes cSizes = {domain.sizes[0] - 1, domain.sizes[1] - 1, domain.sizes[2] - 1};
276
277 Container meshVector(omp_get_max_threads());
278 std::for_each(meshVector.begin(), meshVector.end(), [](auto mesh) { mesh.reserve(MESHSIZE); });
279
280 #pragma omp parallel for
281 for (MCindex cZorder = 0; cZorder < size; ++cZorder) {
282 auto cIndex = core::zorderToPos(cZorder, cSizes);
283 meshVector[omp_get_thread_num()].append(
284 core::triangulateCell(grid(cIndex[0], cIndex[1], cIndex[2]), isoValue)
285 );
286 }
287
288 return meshVector;
289 }

297 template<typename _Ty>
298 struct CheckerTriangulator final : public Triangulator<CheckerMesh<_Ty>> {
299 using Container = typename Triangulator<CheckerMesh<_Ty>>::Container;
300
301 template<class _Data>
302 MC_CONSTEXPR auto operator()(
303 const Grid<_Data>& grid, const typename _Data::Float isoValue
304 ) const {
305 Container meshVector;
306 meshVector.reserve(8);
307 auto size = grid.domain().sizes;
308
309 #pragma omp parallel for collapse(3)
310 for (MCindex p = 0; p < 2; ++p)
311 for (MCindex q = 0; q < 2; ++q)
312 for (MCindex r = 0; r < 2; ++r) {
313 CheckerMesh<_Ty> mesh;
314 mesh.reserve(MESHSIZE);
315
316 for (auto i = p; i < size[0] - 1; i += 2)
317 for (auto j = q; j < size[1] - 1; j += 2)
318 for (auto k = r; k < size[2] - 1; k += 2)
319 mesh.append(core::triangulateCell(grid(i, j, k), isoValue));
320
321 mesh.shrinkToFit();
322 meshVector.push_back(mesh);
323 }
324
325 return meshVector;
326 }
327 };

341MC_NODISCARD auto triangulate(
342 const Grid<_Data>& grid, const typename _Data::Float isoValue, _TriangulationModel&& model
343) noexcept -> decltype(model(grid, isoValue)) {
344 static_assert(
345 is_valid_triangulation_model_v<_TriangulationModel, _Data>,
346 "Handling of undefined triangulation functors (triangulators) is forbidden"
347 );
348
349 api::Report::begin("Triangulating using marching cubes...");
350
351 try {
352 // Checks whether the grid data is appropriate
353 api::error::rtcheck(grid.valid(), api::error::IA, MC_ERROR_MSG("the grid data/domain is not valid"));
354 // Checks whether the triangulation isovalue is a number
355 api::error::rtcheck(!std::isnan(isoValue), api::error::IA, MC_ERROR_MSG("the given isovalue is not a number"));
356
357 auto meshVector = model(grid, isoValue);
358 api::Report::end();
359 return meshVector;
360 }
361 catch (const std::exception& error) { api::error::raise(error); }
362}
