67struct Point final {
69 using Float = _Ty;
70
71 // Data stored in different formats: primitive container and componentwise
72 union {
73 Float v[3];
74 struct { Float x, y, z; };
75 };
76
78 Point(Float X, Float Y, Float Z) noexcept : x(X), y(Y), z(Z) {}
80 Point() noexcept : Point<Float>(Float(0), Float(0), Float(0)) {}
82 Point(const Float* P) : Point<Float>(*P, *(P + 1), *(P + 2)) {}
83
84#if !MC_VERSION_GREATER_THAN(0, 9)
86 Point(const Point<Float>& other) noexcept : Point<Float>(other.v) {}
87
89 Point<Float>& operator=(const Point<Float>& rhs) noexcept {
90 std::copy(rhs.v, rhs.v + 3, v);
91 return *this;
92 }
93#endif // <= v0.9
94
103 MCbool operator==(const Point<Float>& rhs) const noexcept {
104#if MC_VERSION_GREATER_THAN(0, 6)
105 return (*this - rhs).length() < core::epsilon<Float>();
106#else
107 return std::inner_product(v, v + 3, rhs.v, true, std::logical_and<bool>(),
108 [](Float left, Float right) { return std::abs(left - right) < core::epsilon<Float>(); });
109#endif // > v0.6
110 }
111
121 MCbool operator<(const Point<Float>& rhs) const noexcept {
122 if (std::abs(x - rhs.x) > core::epsilon<Float>()) return x < rhs.x;
123 if (std::abs(y - rhs.y) > core::epsilon<Float>()) return y < rhs.y;
124 return z < rhs.z;
125 }
126
134 friend Point<Float> operator*(Float factor, const Point<Float>& rhs) noexcept {
135 return Point<Float>(factor * rhs.x, factor * rhs.y, factor * rhs.z);
136 }
137
143 friend Point<Float> operator*(const Point<Float>& lhs, Float factor) noexcept { return factor * lhs; }
144
153 friend Point<Float> operator+(const Point<Float>& lhs, const Point<Float>& rhs) noexcept {
154 return Point<Float>(lhs.x + rhs.x, lhs.y + rhs.y, lhs.z + rhs.z);
155 }
156
162 friend Point<Float> operator-(const Point<Float>& lhs, const Point<Float>& rhs) noexcept {
163 return lhs + (Float(-1) * rhs);
164 }
165
166#if MC_VERSION_GREATER_THAN(0, 6)
168 auto length() const noexcept { return std::sqrt(std::inner_product(v, v + 3, v, Float(0))); }

224MC_REQUIRE_FLOAT(_Ty)
225struct Domain final : public api::MetaObject {
227 using Float = _Ty;
229 using Point = MC_NAMESPACE::Point<Float>;
234 using MappingRule = std::function<Point(MCindex, MCindex, MCindex)>;
235
236 Point aspect;
237 Point origin;
238 GridSizes sizes;
239
241 Domain() noexcept { clear(); }
242
251 static MappingRule getMappingRule(const Point aspect, const Point origin) noexcept {
252 return [aspect, origin](MCindex i, MCindex j, MCindex k) {
253 return origin + Point(i * aspect.x, j * aspect.y, k * aspect.z);
254 };
255 }
256
261 Point map(MCindex i, MCindex j, MCindex k) const { return getMappingRule(aspect, origin)(i, j, k); }
262
267 MCbool physical() const noexcept {
268 return std::all_of(aspect.v, aspect.v + 3, [](auto value) { return value > 0; }) &&
269 std::equal(origin.v, origin.v + 3, origin.v);
270 }
271
273 MCbool valid() const noexcept {
274 return std::all_of(sizes.begin(), sizes.end(), [](auto size) { return size > 1; });
275 }
276
277 /* \brief Returns \c true, if all sizes are equal to zero. */
278 MCbool empty() const noexcept {
279 return std::all_of(sizes.begin(), sizes.end(), [](auto size) { return size == 0; });
280 }
281
283 MCindex resolution() const noexcept { return sizes[0] * sizes[1] * sizes[2]; }
284
286 MCindex size() const noexcept {
287 return (valid()) ?
288 std::accumulate(sizes.begin(), sizes.end(), MCindex(1),
289 [](MCindex val, MCindex size) { return val * (size - 1); }) : 0;
290 }
291
293 api::MetaCall report() const noexcept override {
294 return [this] {
295 api::information::metamodule(1, "Domain");
296 api::information::metavalue(2, "Aspect ratio",
297 fmt::format(fg(fmt::color::cadet_blue), "({})", fmt::join(aspect.v, ", "))
298 );
299 api::information::metavalue(2, "Origin",
300 fmt::format(fg(fmt::color::cadet_blue), "({})", fmt::join(origin.v, ", "))
301 );
302 api::information::metavalue(2, "Resolution",
303 fmt::format(fg(fmt::color::cadet_blue), "{}", fmt::join(sizes, " x "))
304 );
305 api::information::metavalue(2, "Number of points", fmt::format(fg(fmt::color::cadet_blue), "{}", resolution()));
306 api::information::metavalue(2, "Number of cubes", fmt::format(fg(fmt::color::cadet_blue), "{}", size()));
307 api::information::metamodule(2, "Bounding box");
308 api::information::metavalue(3, "bottom", fmt::format(fg(fmt::color::cadet_blue), "({})",
309 fmt::join(origin.v, ", "))
310 );
311 api::information::metavalue(3, "top", fmt::format(fg(fmt::color::cadet_blue), "({})",
312 fmt::join((origin + Point(aspect.x * sizes[0], aspect.y * sizes[1], aspect.z * sizes[2])).v, ", "))
313 );
314 };
315 }
316
321 void clear() noexcept {
322 aspect = Point();
323 std::fill_n(origin.v, 3, core::NaN<Float>());
324 sizes.fill(0);
325 }
326 // !public

341 inline MCindex posToZorder(const GridSizes pos, const GridSizes sizes) noexcept {
342 return pos[0] + (pos[1] + pos[2] * sizes[1]) * sizes[0];
343 }
344
352 inline GridSizes zorderToPos(const MCindex zorder, const GridSizes sizes) noexcept {
353 GridSizes pos = {-1, -1, -1};
354
355 if (std::count(sizes.begin(), sizes.end(), 0) == 0) {
356 pos[0] = zorder % sizes[0];
357 auto zorderReduced = (zorder - pos[0]) / sizes[0];
358 pos[1] = zorderReduced % sizes[1];
359 pos[2] = (zorderReduced - pos[1]) / sizes[1];
360 }
361
362 return pos;
363 }
364
375 template<typename _DifferenceType, MC_ENABLE_IF(std::is_signed_v<_DifferenceType>)>
376 _DifferenceType distance(const GridSizes left, const GridSizes right, const GridSizes sizes) noexcept {
377 auto leftZorder = static_cast<_DifferenceType>(posToZorder(left, sizes));
378 auto rightZorder = static_cast<_DifferenceType>(posToZorder(right, sizes));
379 return leftZorder - rightZorder;
380 }

507template<typename _Ty>
508struct RawData final : public impl::_RawData_base<_Ty> {
510 using Float = typename impl::_RawData_base<_Ty>::Float;
512 using Initializer = typename impl::_RawData_base<Float>::Initializer;
514 using Holder = typename std::tuple_element<0, Initializer>::type;
515
517 RawData(const Initializer& init = Initializer()) noexcept { initialize(init); }
519 ~RawData() override { this->clear(); }
520
521 const String type() const noexcept override { return "raw_data_manager"; }
522
536 MCbool initialize(const Initializer& init) noexcept override {
537 Holder holder;
538 GridSizes sizes;
539 std::tie(holder, sizes) = init;
540
541 auto size = holder.size();
542 if (size && size == std::accumulate(sizes.begin(), sizes.end(), MCindex(1), std::multiplies<MCindex>())) {
543 this->_handler = [holder, sizes](MCindex i, MCindex j, MCindex k) -> Float {
544 return holder[core::posToZorder({i, j, k}, sizes)];
545 };
546
547 auto typ = type();
548 auto min = *std::min_element(holder.begin(), holder.end());
549 auto max = *std::max_element(holder.begin(), holder.end());
550 auto avg = std::accumulate(holder.begin(), holder.end(), Float(0)) / size;
551
552 this->_report = [typ, min, max, avg] {
553 api::information::metamodule(1, "Data");
554 api::information::metavalue(2, "Type", fmt::format(fg(fmt::color::cadet_blue), "{}", typ));
555 api::information::metavalue(2, "Minimum value", fmt::format(fg(fmt::color::cadet_blue), "{:+f}", min));
556 api::information::metavalue(2, "Maximum value", fmt::format(fg(fmt::color::cadet_blue), "{:+f}", max));
557 api::information::metavalue(2, "Average value", fmt::format(fg(fmt::color::cadet_blue), "{:+f}", avg));
558 };
559
560 return true;
561 }
562 else {
563 this->clear();
564 return false;
565 }
566 }

619 MCbool initialize(const Initializer& init) noexcept override {
620 Holder holder;
621 Point<Float> aspect, origin;
