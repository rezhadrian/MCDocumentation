117 inline void rtcheck(
118 MCbool statement = true, MCint code = RT,
119 const String& message = "meaningful error message", const MetaCall& action = MetaCall()
120 ) {
121 using namespace std;
122 if (!statement) {
123 switch (code) {
124 case RT: throw runtime_error{message.c_str()};
125 case IA: throw invalid_argument{message.c_str()};
126 case OR: throw out_of_range{message.c_str()};
127 case OF: throw overflow_error{message.c_str()};
128 case LE: throw logic_error{message.c_str()};
129 default: throw runtime_error{"Unknown error"};
130 }
131 }
132 // If the statement is a final one, a quick access to further actions is proposed in case of the truth
133 else if (action) action();
134 }
