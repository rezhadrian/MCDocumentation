88 inline static StringVector split(const String& string, const String& separator = "") noexcept {
89 StringVector result;
90 auto sepLength = separator.length();
91
92 if (sepLength < string.length()) {
93 if (sepLength > 0) {
94 MCindex found = -sepLength;
95
96 std::vector<MCindex> seps;
97 seps.reserve(51); // 50 substrings
98 seps.push_back(found);
99
100 while (true) {
101 found = string.find(separator, found + sepLength);
102 if (found < -1) seps.push_back(found);
103 else break;
104 }
105 seps.push_back(string.length());
106 seps.shrink_to_fit();
107
108 auto size = seps.size() - 1;
109 result.reserve(size);
110
111 for (MCindex i = 0; i < size; ++i) {
112 auto count = seps[i + 1] - seps[i] - sepLength;
113 if (count > 0) result.push_back(string.substr(seps[i] + sepLength, count));
114 }
115 result.shrink_to_fit();
116 }
117 else {
118 result.resize(string.length());
119 std::copy(string.begin(), string.end(), result.begin());
120 }
121 }
122 else result.push_back(string);
123
124 return result;
125 }

258 void load(fmt::file& from, Grid& grid) const override {
270 Domain domain;
271 Holder holder;
272
273 MCbool header_is_filled = false;
274 MCindex header = 9;
275
276 std::vector<MCindex> index_holder;
277 index_holder.reserve(3);
278
279 std::vector<Float> physical_data;
280 physical_data.reserve(6);
281
282 MCindex rread; // Real byte data read
283 fmt::memory_buffer mb; // fmt:: internal buffer
284 mb.resize(file::BUFSIZE);
285 String sb = ""; // String buffer (algorithmically preferred, but may be slow)
286
287 do {
288 rread = file::SpecialMethods::read(from, mb);
289 sb.append(mb.begin(), rread);
290 std::replace_if(sb.begin(), sb.end(), filter(), ' ');
291 auto values = file::SpecialMethods::split(sb, " ");
292
293 // If the last read element is not an empty space, then
294 // the number was cut and should be extended with the byte
295 // data from the next buffer
296 if (rread && sb.back() != ' ') {
297 sb = values.back();
298 values.pop_back();
299 }
300 else sb.clear();
301
302 for (auto& value : values) {
303 if (header_is_filled) {
304 if (index_holder.size() < 3) {
305 index_holder.push_back(std::strtoull(value.c_str(), NULL, 10));
306 }
307 else {
308 holder[core::posToZorder(
309 {index_holder[0], index_holder[1], index_holder[2]},
310 domain.sizes
311 )] = std::stold(value);
312 index_holder.clear();
313 }
314 }
315 else { // If 9 values are not read, then the header is not filled yet
316 if (header > 6) {
317 index_holder.push_back(std::strtoull(value.c_str(), NULL, 10));
318 header--;
319 continue;
320 }
321
322 if (header == 6)
323 {
324 domain.sizes[0] = index_holder[0];
325 domain.sizes[1] = index_holder[1];
326 domain.sizes[2] = index_holder[2];
327 holder.resize(domain.resolution(), core::NaN<Float>());
328 index_holder.clear();
329 }
330
331 if (header) {
332 physical_data.push_back(std::stold(value));
333 if (header == 1) {
334 domain.aspect = {physical_data[0], physical_data[1], physical_data[2]};
335 domain.origin = {physical_data[3], physical_data[4], physical_data[5]};
336 header_is_filled = true;
337 }
338 header--;
339 }
340 }
341 }
342 } while (rread);
343
344 grid.setDomain(domain);
345 grid.setData(Data(std::make_tuple(holder, domain.sizes)));
346 }

466 }

469 if (aspect.size() == 3) as_aspect = false;

473 if (origin.size() == 3) as_origin = false;
474 }
475
476 if (index_holder.size() == 3 && aspect.size() == 3 && origin.size() == 3) {
477 domain.aspect = aspect.data();
478 domain.origin = origin.data();
479 std::copy(index_holder.begin(), index_holder.end(), domain.sizes.begin());
480 holder.reserve(domain.resolution());
481 aspect.clear();
482 origin.clear();
483 index_holder.clear();
484 }
485 }
486 }
487 } while (rread);
488
489 grid.setDomain(domain);
